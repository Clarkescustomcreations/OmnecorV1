# ============================================================
# File: blender_bridge.py
# Purpose:
#   Execute external Python scripts inside Blender's
#   headless/background environment for the Omnecor HMCI
#   Hardware Integration Layer.
#
# Usage:
#   blender -b -P blender_bridge.py -- --script my_generator.py
#
# Notes:
#   - The "--" separator is REQUIRED when passing custom args
#     to Blender Python scripts.
#   - The target script executes INSIDE Blender's Python runtime.
# ============================================================

import argparse
import json
import os
import runpy
import sys
import traceback


def emit(status, message, extra=None):
    """
    Emit structured JSON output for Omnecor backend parsing.
    """
    payload = {
        "status": status,
        "message": message
    }

    if extra:
        payload.update(extra)

    print(json.dumps(payload), flush=True)


def parse_blender_args():
    """
    Blender passes its own CLI args before "--".
    Everything after "--" belongs to our script.
    """
    argv = sys.argv

    if "--" in argv:
        argv = argv[argv.index("--") + 1:]
    else:
        argv = []

    parser = argparse.ArgumentParser(
        description="Omnecor Blender Headless Script Executor"
    )

    parser.add_argument(
        "--script",
        required=True,
        help="Path to Python script to execute inside Blender"
    )

    return parser.parse_args(argv)


def main():
    args = parse_blender_args()

    script_path = os.path.abspath(args.script)

    emit("info", f"Executing Blender script: {script_path}")

    if not os.path.exists(script_path):
        emit(
            "error",
            "Script file does not exist",
            {"script": script_path}
        )
        sys.exit(1)

    try:
        # Execute the target script inside Blender context
        # using the current global namespace.
        runpy.run_path(script_path, run_name="__main__")

        emit(
            "success",
            "Script executed successfully",
            {"script": script_path}
        )

        sys.exit(0)

    except Exception as e:
        emit(
            "failure",
            "Script execution failed",
            {
                "script": script_path,
                "error": str(e),
                "traceback": traceback.format_exc()
            }
        )

        sys.exit(1)


if __name__ == "__main__":
    main()
